# Validate generated epic structure
validate_generated_epic() {
    local epic_dir="$1"
    
    # Check for epic.md
    if [[ ! -f "$epic_dir/epic.md" ]]; then
        echo "âŒ Missing epic.md in generated epic"
        return 1
    fi
    
    # Check for task files
    local task_count=$(ls -1 "$epic_dir"/[0-9][0-9][0-9].md 2>/dev/null | wc -l)
    if [[ $task_count -eq 0 ]]; then
        echo "âŒ No task files generated in epic"
        return 1
    fi
    
    # Validate task file frontmatter
    local invalid_tasks=0
    for task_file in "$epic_dir"/[0-9][0-9][0-9].md; do
        [[ -f "$task_file" ]] || continue
        if ! validate_yaml_frontmatter "$task_file" 2>/dev/null; then
            echo "âŒ Invalid task file: $task_file"
            ((invalid_tasks++))
        fi
    done
    
    if [[ $invalid_tasks -gt 0 ]]; then
        echo "âŒ Found $invalid_tasks invalid task files"
        return 1
    fi
    
    echo "âœ… Generated epic structure is valid ($task_count tasks)"
    return 0
}
